if($(".article-cat div a").length){

  $(".article-cat div a").each(function () {
    //alert(this.href);
  
    let thelink;
  
    thelink = $(this).attr('href');
  
    thelink = thelink.replace("/category", ""); 
  
    $(this).attr('href', thelink);
  });
  
}
